CREATE VIEW Changes (IdEmployee, OperationType, Description, OperationDate)
AS SELECT IdEmployee, TypeOperation, Description, OperationDate
FROM Log;

AS SELECT IdEmployee, COUNT(IdEmployee)
FROM Changes
Group BY IdEmployee;
